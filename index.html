JavaScript示例

import OpenAI from 'openai';

const openai = new OpenAI({
  baseURL: 'https://zzzzapi.com/v1', // 替换为我们的服务地址
  apiKey: 'sk-xxxx', // 使用您的API密钥
});

async function main() {
  const stream = await openai.chat.completions.create({
    model: 'gpt-4o',
    messages: [{ role: 'user', content: '请介绍一下ChatGPT API的优势' }],
    stream: true,
  });
  
  for await (const chunk of stream) {
    process.stdout.write(chunk.choices[0]?.delta?.content || '');
  }
}

main();

import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.service.OpenAiService;

public class OpenAiExample {
    public static void main(String[] args) {
        // 创建OpenAI服务实例，使用我们的API密钥
        OpenAiService service = new OpenAiService("sk-xxxx");
        
        // 创建聊天完成请求
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("gpt-4o")
                .addMessage(new ChatMessage("user", "请介绍一下ChatGPT API的优势"))
                .build();
                
        // 发送请求并处理响应
        service.createChatCompletion(request)
                .getChoices()
                .forEach(choice -> {
                    System.out.println(choice.getMessage().getContent());
                });
    }
}
